不解压直接查看tar包内容
1. file.tar.gz
gzip -dc file.tar.gz | tar tvf -

2. file.tar.bz2
bzip2 -dc file.tar.bz2 |tar tvf -

3. file.tar
tar tvf file.tar

find ./ -name filename

tail -n 20
